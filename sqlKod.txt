Maturski rad kod :

Tekst za maturski rad :

Potrebno je napraviti bazu podataka za Košarkašku Ligu Srbije.
Liga se sastoji iz 16 timova.

Svaki tim ima svoj naziv, logotip, adresu, kontakt informacije(telefon, email) i dvoranu.
Tim ima jednog ili više zaposlenih za koje se pamti id, ime, prezime i slika, a takođe Zaposleni tabela sadrži i foreign key kojim se zna da li je zaposleni glavni trener,pomoćni trener,lekar,pratilac ili službeni predstavnik (za tabelu o kategoriji zaposlenog se pamti id i naziv te kategorije).
Tokom sezone se prati plasman timova stoga je potrebno imati i entitet o rang listi timova koji može sadržati informacije o jednom ili više timova i u kojoj se prati trenutno kolo sezone, pobede/porazi tima,dati koševi/promašeni koševe (KD:KP),razlika datih i promašenih koševa (KR) i broj bodova.
Jedan tim može imati jednog ili više igrača. Za svakog igrača potrebno je imati njegov evidencioni broj, ime, prezime, broj dresa, godište, visinu, poziciju i sliku.
Timovi mogu odigrati jednu ili više utakmica i jedna utakmica može biti odigrana od strane 2 tima.

Potrebno je pratiti utakmice u kojima se timovi takmiče, stoga entitet Utakmica pamti rezulat, delegat, broj gledalaca, datum kada je utakmica odigrana kao i strane ključeve (id-jeve za domaći i protivnički tim).

Nakon utakmice se treba znati i statistika svakog igrača sa iste, zato je potrebno imati entitet o Statistici igrača na utakmici za koji će se gledati broj poena igrača,broj pogodaka za 1,2 i 3 poena, procenat šuteva kao i skokovi, asistencije, blokovi i broj minuta koji je igrač proveo na parketu.

Igrač može odigrati jednu ili više utakmica i jedna utakmica može biti odigrana od strane više igrača.

Za svaku utakmicu se mora znati dvorana u kojoj je odigrana pa tako u jednoj dvorani može da se odigra jedna ili više utakmica. Za dvoranu se pamti njen id i naziv.

Sudija može da sudi na jednoj ili više utakmica i Utakmica može da bude suđena od strane jednog ili više sudija.
Entitet Sudija beleži id, ime, prezime, mesto iz kojeg sudija dolazi i sliku.




PRAVLJENJE TABELA :

CREATE TABLE KATEGORIJE_ZAP (
    id NUMBER(7) PRIMARY KEY,
    naziv VARCHAR2(255) NOT NULL
);

CREATE TABLE RANG_LISTA_TIMOVA (
    id NUMBER(4) PRIMARY KEY,
    ut NUMBER(4) NOT NULL,
    p_i VARCHAR2(15) NOT NULL,
    kd_kp VARCHAR2(15) NOT NULL,
    kr VARCHAR2(10) NOT NULL,
    bodovi VARCHAR2(7) NOT NULL
);

CREATE TABLE DVORANE_UTAKMICA (
    id VARCHAR2(3) PRIMARY KEY,
    naziv VARCHAR2(50) NOT NULL
);

CREATE TABLE SUDIJE (
    id NUMBER(5) PRIMARY KEY,
    ime VARCHAR2(20) NOT NULL,
    prezime VARCHAR2(20) NOT NULL,
    mesto VARCHAR2(30) NOT NULL,
    slika VARCHAR2(150) NOT NULL
);

CREATE TABLE TIMOVI (
    id NUMBER(2) PRIMARY KEY,
    naziv VARCHAR2(50) NOT NULL,
    logo VARCHAR2(150) NOT NULL,
    adresa VARCHAR2(50) NOT NULL,
    telefon VARCHAR2(30) NOT NULL,
    email VARCHAR2(50) NOT NULL,
    dvorana VARCHAR2(50) NOT NULL,
    rlt NUMBER(4) REFERENCES RANG_LISTA_TIMOVA(id)
);

CREATE TABLE IGRACI (
    eb VARCHAR2(10) PRIMARY KEY,
    ime VARCHAR2(25) NOT NULL,
    prezime VARCHAR2(30) NOT NULL,
    broj_dresa NUMBER(3),
    godiste NUMBER(4) NOT NULL,
    visina VARCHAR2(10) NOT NULL,
    pozicija VARCHAR2(30) NOT NULL,
    slika VARCHAR2(150) NOT NULL,
    TIM_ID NUMBER(2) REFERENCES TIMOVI(id)
);

CREATE TABLE ZAPOSLENI (
    id VARCHAR2(100) PRIMARY KEY,
    ime VARCHAR2(12) NOT NULL,
    prezime VARCHAR2(20) NOT NULL,
    slika VARCHAR2(150) NOT NULL,
    TIM_ID NUMBER(2) REFERENCES TIMOVI(id),
    KAT_ID NUMBER(7) REFERENCES KATEGORIJE_ZAP(id)
);

CREATE TABLE UTAKMICE (
    id_domacina NUMBER(2),
    id_protivnika NUMBER(2),
    rezultat VARCHAR2(7) NOT NULL,
    delegat VARCHAR2(30) NOT NULL,
    broj_gledalaca NUMBER(4) NOT NULL,
    datum DATE NOT NULL,
    id_dvorane VARCHAR2(3) REFERENCES DVORANE_UTAKMICA(id),
    PRIMARY KEY (id_domacina, id_protivnika),
    CONSTRAINT fk_domacin FOREIGN KEY (id_domacina) REFERENCES TIMOVI(id),
    CONSTRAINT fk_protivnik FOREIGN KEY (id_protivnika) REFERENCES TIMOVI(id)
);

CREATE TABLE STAT_I_NA_U (
    id NUMBER(5),
    poeni NUMBER(10) NOT NULL,
    jedanP VARCHAR2(10) NOT NULL,
    jedanP_procenti VARCHAR2(5) NOT NULL,
    dvaP VARCHAR2(10) NOT NULL,
    dvaP_procenti VARCHAR2(5) NOT NULL,
    triP VARCHAR2(10) NOT NULL,
    triP_procenti VARCHAR2(5) NOT NULL,
    skokovi VARCHAR2(15) NOT NULL,
    lop VARCHAR2(10) NOT NULL,
    asistencije VARCHAR2(5) NOT NULL,
    blokovi VARCHAR2(5) NOT NULL,
    lg VARCHAR2(5) NOT NULL,
    minuti NUMBER(5) NOT NULL,
    igrac_id VARCHAR2(10) REFERENCES IGRACI(eb),
    utakmica_id_domacina NUMBER(2),
    utakmica_id_protivnika NUMBER(2),
    PRIMARY KEY (id,igrac_id, utakmica_id_domacina, utakmica_id_protivnika),
    CONSTRAINT fk_utakmica_domacin FOREIGN KEY (utakmica_id_domacina, utakmica_id_protivnika) REFERENCES UTAKMICE(id_domacina, id_protivnika)
);

CREATE TABLE SUDIJE_UTAKMICE (
    id_sudije NUMBER(5),
    id_utakmice_domacina NUMBER(2),
    id_utakmice_protivnika NUMBER(2),
    PRIMARY KEY (id_utakmice_domacina, id_utakmice_protivnika, id_sudije),
    CONSTRAINT fk_utakmica FOREIGN KEY (id_utakmice_domacina, id_utakmice_protivnika) REFERENCES UTAKMICE(id_domacina, id_protivnika),
    CONSTRAINT fk_sudija FOREIGN KEY (id_sudije) REFERENCES SUDIJE(id)
);

NAKON OVOGA SAM KUCAO OVO :

ALTER TABLE STAT_I_NA_U DROP CONSTRAINT fk_utakmica_domacin;
ALTER TABLE SUDIJE_UTAKMICE DROP CONSTRAINT fk_utakmica;
ALTER TABLE STAT_I_NA_U DROP COLUMN utakmica_id_protivnika;
ALTER TABLE SUDIJE_UTAKMICE DROP COLUMN id_utakmice_protivnika;
ALTER TABLE STAT_I_NA_U DROP PRIMARY KEY;
ALTER TABLE STAT_I_NA_U ADD PRIMARY KEY (id,igrac_id, utakmica_id_domacina);
ALTER TABLE SUDIJE_UTAKMICE DROP PRIMARY KEY;
ALTER TABLE SUDIJE_UTAKMICE ADD PRIMARY KEY (id_utakmice_domacina, id_sudije);
ALTER TABLE STAT_I_NA_U
    ADD CONSTRAINT fk_utakmica_domacin FOREIGN KEY (utakmica_id_domacina) REFERENCES UTAKMICE(id_domacina);
ALTER TABLE SUDIJE_UTAKMICE
    ADD CONSTRAINT fk_utakmica FOREIGN KEY (id_utakmice_domacina) REFERENCES UTAKMICE(id_domacina);
